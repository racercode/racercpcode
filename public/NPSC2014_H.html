const int maxn = 1e6 + 5 ;
const int mod = 1e9 + 7 ;
const int INF = 1e9 + 9 ;
int dis[maxn],c[maxn],n,m,t,p;
void dij()
{
	dis[0]=0;
	priority_queue<pii,vector<pii>,greater<pii> >pq;
	pq.push({0,0});
	while(!pq.empty())
	{
		auto i=pq.top();
		pq.pop();
		if(i.f>dis[i.s]) continue;
		for(int j=2;j<=n;j++)
		{
			if(i.f+c[j]<dis[(i.s+c[j])%p])
			{
				dis[(i.s+c[j])%p]=i.f+c[j];
				pq.push({i.f+c[j],(i.s+c[j])%p});
			}
		}
	}
}
signed main()
{
	iofaster;
	cin>>t;
	while(t--)
	{
		memset(dis,0x3f,sizeof(dis));
		cin>>n>>m;
		for(int i=1;i<=n;i++) cin>>c[i];
		sort(c+1,c+1+n);
		p=c[1];
		dij();
		while(m--)
		{
			int x;
			cin>>x;
			if(dis[x%p]<=x) putchar('Y');
			else putchar('N');
		}
		putchar('\n');
	}
}
